Part 1: Getting a Preliminary GUI (10 mins)

Processing uses Java and has two main functions:

  • setup() â€“ runs once at the start.
  • draw() â€“ runs repeatedly, allowing animations.

We will use the ControlP5 library to create GUI elements.

import controlP5.*; // import ControlP5 library

ControlP5 cp5; // create ControlP5 object
PFont myFont;  // font object

void setup(){
    size(300, 400); // window size
    myFont = createFont("Arial-Bold-32", 32); // bold font, size 32
    textFont(myFont); // apply font
}

void draw(){
    background(50, 70, 100); // background color
    fill(255); // white text
    text("GUITAR HERO", 50, 50); // text position
}

Part 2: Adding a Button

import controlP5.*; // import ControlP5 library

ControlP5 cp5; // create ControlP5 object
PFont myFont;  // font object

void setup(){
    size(300, 400); // window size
    myFont = createFont("Arial-Bold-32", 32);
    textFont(myFont);

    cp5 = new ControlP5(this); // initialize ControlP5

    // Add START button
    cp5.addButton("START")
       .setPosition(80, 150)
       .setSize(140, 70)
       .setFont(myFont)
       .setColorCaptionLabel(color(255))
       .setColorBackground(color(0, 150, 255))
       .setColorActive(color(0, 200, 255));
}

void draw(){
    background(50, 70, 100);
    fill(255);
    textFont(myFont);
    text("GUITAR HERO", 40, 60);
}

Part 3: Adding Circle Animation

import controlP5.*; // import ControlP5 library

ControlP5 cp5; 
PFont myFont;  
float circleY = 0;

void setup(){
    size(300, 400);
    myFont = createFont("Arial-Bold-32", 32);
    textFont(myFont);

    cp5 = new ControlP5(this);

    cp5.addButton("START")
       .setPosition(100, 120)
       .setSize(100, 50)
       .setFont(myFont)
       .setColorCaptionLabel(color(255))
       .setColorBackground(color(0, 150, 255))
       .setColorActive(color(0, 200, 255));
}

void draw(){
    background(50, 70, 100);

    fill(255, 100, 100); // red circle
    ellipse(150, circleY, 50, 50); 
    circleY += 2; 
    if(circleY > height){ circleY = 0; }

    fill(255);
    textFont(myFont);
    text("GUITAR HERO", 50, 50);
}
